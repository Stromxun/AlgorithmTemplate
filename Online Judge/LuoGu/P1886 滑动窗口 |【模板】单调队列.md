### 洛谷 P1886 [P1886 滑动窗口 /【模板】单调队列](https://www.luogu.com.cn/problem/P1886) 
##### 知识点： 单调队列

```cpp
#include<bits/stdc++.h>

using namespace std;
const int N=1000005;
int a[N];
deque<int>q;
int main(){
	int n,m;
	scanf("%d%d",&n,&m);
	for(int i=1;i<=n;i++) scanf("%d",&a[i]);
	for(int i=1;i<=n;i++){
		while(!q.empty()&&a[q.back()]>a[i]) q.pop_back();
		q.push_back(i);
		if(i>=m){
			while(!q.empty()&&q.front()<=i-m) q.pop_front();//不在窗口范围内，退队首
			printf("%d ",a[q.front()]);
		}
	}
	printf("\n");
	q.clear();
	for(int i=1;i<=n;i++){
		while(!q.empty()&&a[q.back()]<a[i]) q.pop_back();
		q.push_back(i);
		if(i>=m){
			while(!q.empty()&&q.front()<=i-m) q.pop_front();//不在窗口范围内，退队首
			printf("%d ",a[q.front()]);
		}
	}
	printf("\n");
	return 0;
}
```
